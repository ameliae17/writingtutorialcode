import java.util.Scanner;

public class PortfolioTech
{

    public static void main(String[] args)
    {
        Scanner scanner = null;

        try
        {
            scanner = new Scanner(System.in);
            System.out.println("What's your question?");
            String input = scanner.nextLine();

            if(input.contains("What is good writing?"))
            {
                System.out.printf("'Good writing' can take on many forms and is very subjective. To some, 'good writing' is uniform and traditional––something that sounds familiar. Because it's familiar, it has to be correct, which is untrue. Instead, 'good writing' " +
                                  "is writing which guides you in a logical way and allows you to form your own opinion to be able to agree, disagree, or neither. ");
            }

            else if(input.contains("What are effective comments?"))
            {
                System.out.printf("Effective comments noticeably come in three forms: evaluating coaching, or reacting comments.");
            }

            else
            {
                System.out.printf("I didn't understand your question. Please use a question from the list above!");
            }

        }

        catch(Exception e)
        {
            System.out.println("An error occurred: " + e.getMessage());
        }

        finally
        {
            if(scanner!= null)
            {
                scanner.close();
            }
        }
    }
}
